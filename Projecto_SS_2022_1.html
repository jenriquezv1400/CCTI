; ******************************************************
; BASIC .ASM template file for AVR
; ******************************************************
.include "C:\VMLAB\include\m8535def.inc"
; Define here the variables
.def temp =r16
;Registros para lazos de perdida de tiempo:
;r17
;r18
;r19
;Registros para Varaibles
.def DATO =r20
.def COMPARA =r21
.def SENSOR1 =r22
.def SENSOR2 =r23
.def SENSOR3 =r24
; Define here Reset and interrupt vectors, if any
;
reset:
rjmp start
reti ; Addr $01
reti ; Addr $02
reti ; Addr $03
reti ; Addr $04
reti ; Addr $05
reti ; Addr $06 Use 'rjmp myVector'
reti ; Addr $07 to define a interrupt vector
reti ; Addr $08
reti ; Addr $09
reti ; Addr $0A
reti ; Addr $0B This is just an example
reti ; Addr $0C Not all MCUs have the same
reti ; Addr $0D number of interrupt vectors
rjmp INT_ADC ; Addr $0E
reti ; Addr $0F
reti ; Addr $10
; Program starts here after Reset
;
start:
;3)cONFIGURA pILA
LDI TEMP,$02
OUT SPH,TEMP
LDI TEMP,$5f
OUT SPL,TEMP
;4) cONFIGURA PUERTO DE SALIDA
LDI TEMP,$FF
OUT DDRD,TEMP; Puerto D Como salida
;5) CONFIGURA ADC
LDI TEMP,0B01100000
OUT ADMUX,TEMP
LDI TEMP,0B10001100
OUT ADCSRA,TEMP
;6) VALORES INCIALES
CLR DATO
OUT PORTC,DATO
OUT PORTD,DATO
;7) HABILITAR INTERRUPCIÓN
SEI
;8) INICIAR CONVERSIÓN
SBI ADCSRA,6
;===================================
;==========Loop principal============
;===================================
forever:
; IN DATO,PIND
; OUT PORTB,DATO
nop ; Infinite loop.
nop ; Define your main systembehaviour here
rjmp forever
;===================================
;============FUNCIONES==============
;===================================
INT_ADC:
LDI TEMP,0B01100000 ;Estás leyendo puerto 0 ?
IN COMPARA,ADMUX
CP TEMP,COMPARA
BREQ Sensor_2
LDI TEMP,0B01100001 ;Estás leyendo puerto 1 ?
IN COMPARA,ADMUX
CP TEMP,COMPARA
BREQ Sensor_3
;aCTIVACIÓN Y DESACTIVACIÓN Bomba
LDI TEMP,0B01100000
OUT ADMUX,TEMP
IN SENSOR1,ADCH
LDI TEMP,0B11100001;900 DECIMAL
CP TEMP,SENSOR1; SI TEMPORAL ES mayor QUE SENSOR 1 SALTA SI NO NO SALTES
BRSH IncHum
LDI TEMP,0B11111010; 1000 DECIMAL
CP SENSOR1,TEMP; SI TEMPORAL ES menor QUE SENSOR 1 SALTA SI NO NO SALTES
BRSH ApagaBomb
SBI PORTD,2 ; SET BIT iN
rjmp Sal
ApagaBomb:
CBI PORTD,2 ; cLEAR BIT iN
rjmp Sal
IncHum:
SBI PORTD,2; prende el pin 3
rjmp Sal
Sensor_2:
;aCTIVACIÓN Y DESACTIVACIÓN Foco
LDI TEMP,0B01100001
OUT ADMUX,TEMP
IN SENSOR2,ADCH
LDI TEMP,0B01100100; 400 DECIMAL
CP TEMP,SENSOR2; SI TEMPORAL ES mayor QUE SENSOR 2 SALTA SI NO NO SALTES
BRSH SinLuz
LDI TEMP,0B11100001; 900 DECIMAL
CP SENSOR2,TEMP; SI TEMPORAL ES menor QUE SENSOR 2 SALTA SI NO NO SALTES
BRSH ApagaLuz
SBI PORTD,3 ; SET BIT iN
rjmp Sal
ApagaLuz:
CBI PORTD,3 ; cLEAR BIT iN
rjmp Sal
SinLuz:
SBI PORTD,3; prende el pin 3
rjmp Sal
Sensor_3:
;aCTIVACIÓN Y DESACTIVACIÓN Resistencia
LDI TEMP,0B01100010; pina A1 Adc
OUT ADMUX,TEMP
IN SENSOR3,ADCH
LDI TEMP,0B00001010; ~< 40 DECIMAL
CP TEMP,SENSOR3; SI TEMPORAL ES mayor QUE SENSOR 1 SALTA SI NO NO SALTES
BRSH IncTemp
LDI TEMP,0B00001110; >~ 57 DECIMAL
CP SENSOR3,TEMP; SI TEMPORAL ES menor QUE SENSOR 1 SALTA SI NO NO SALTES
BRSH ApagaRes
SBI PORTD,4 ; SET BIT iN
rjmp Sal
ApagaRes:
CBI PORTD,4 ; cLEAR BIT iN
rjmp Sal
IncTemp:
SBI PORTD,4; prende el pin 3
rjmp Sal
Sal:
RCALL Pausa1_4s
SBI ADCSRA,6 ;8) INICIAR CONVERSIN
RETI
;======== Retardo por lazos de pérdida de tiempo ===============
Pausa1_4s: ;RET; PARA SIMULACIÓN SE QUEDA
LDI r17,$A7
Loop00:
LDI r18,$02
Loop01:
LDI r19,$F8
Loop02:
DEC r19
BRNE Loop02
DEC r18
BRNE Loop01
DEC r17
BRNE Loop00
NOP
RET
